$m=128;

$count=0;

echo "<th>Product</th>";

foreach ($a as $value) {
  	$b=$m/$value;
  	$c=$value-1;
  	$d=$b-1;
  	$F=(1+$c)*(1+$d);
    $Sum+=$F;
    $count++;
    echo "<tr>";

    echo "<td>$Sum</td>";
    echo "</tr>";
}
echo "</table>";
echo "The average value is: ".$Sum/$count;
function Divisors($n) {
  $cart=array();
  for($i = 1; $i <= $n; $i++) {
    if($n%$i == 0)
     $cart[] = $i; 
  }
  return $cart;
}
