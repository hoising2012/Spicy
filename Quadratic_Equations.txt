$Alpha=1;
$Beta=-1;
$Gamma=-1;
$Q=QE($Alpha,$Beta,$Gamma);
for($i=0; $i<=count($Q)-1; $i++){
	if($i==0){
    	echo "x_0: ".$Q[$i].",";
    }else{
    	echo "x_1: ".$Q[$i];
    }
    echo "\n";
}
function QE($a, $b, $c){
        $root_1 = ((-1)*$b + sqrt(pow($b,2) - 4*$a*$c))/(2*$a);
        $root_2 = ((-1)*$b - sqrt(pow($b,2) - 4*$a*$c))/(2*$a);
		return array($root_1,$root_2);
}
